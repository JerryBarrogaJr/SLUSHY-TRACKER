async function openSaleModal(teamId, memberId, saleId = null) { if (isMutatingLock) return; activeSaleContext = { teamId, memberId, saleId }; let appData = await db._getRaw(); let mObj = appData.members.find(m => m.id === memberId); document.getElementById("targetMemberDisplay").innerText = mObj ? mObj.name : 'Unknown'; if (saleId) { document.getElementById("modalTitle").innerText = "✏️ Edit Sale Entry"; let target = appData.sales.find(s => s.id === saleId); if (target) { document.getElementById("saleModelSelect").value = target.model_name || ""; document.getElementById("saleFan").value = target.fan_name; document.getElementById("saleAmount").value = target.amount; document.getElementById("saleDate").value = target.timestamp.split('T')[0]; document.getElementById("saleTime").value = target.timestamp.split('T')[1].substring(0, 5); } } else { // FIX: Dynamically compute local timezone-safe values on the fly document.getElementById("modalTitle").innerText = "💰 Record New Sale"; document.getElementById("saleModelSelect").value = ""; document.getElementById("saleFan").value = ""; document.getElementById("saleAmount").value = ""; // Auto-populate with current actual date and time const now = new Date(); const localDate = now.toLocaleDateString('sv-SE'); // Outputs perfectly formatted YYYY-MM-DD const localTime = now.toTimeString().split(' ')[0].substring(0, 5); // Outputs cleanly formatted HH:MM document.getElementById("saleDate").value = localDate; document.getElementById("saleTime").value = localTime; } openModal("saleModal"); }
